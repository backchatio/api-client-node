1 var Swagger = require('./swagger'),
  2     util = require('util'),
  3     events = require('events'),
  4     _ = require('underscore');
  5 
  6 /**
  7  * The default BackChat.io API resource discovery URL.
  8  *
  9  * @private
 10  */
 11 var BackchatApiUrl = 'https://api.backchat.io/1/swagger';
 12 
 13 /**
 14  * Handler for the specific reponse format returned by the BackChat.io API.
 15  *
 16  * @private
 17  * @constructor
 18  */
 19 var ResponseHandler = function(meta) {
 20   this.meta = meta;
 21   this.handle = function(error, response, body) {
 22     var errors, data, json, code = response.statusCode;
 23     if (_.include([401, 504], code)) {
 24       errors = [{message:body}];
 25     } else if (code === 204) {
 26       data = {};
 27     } else {
 28       json = JSON.parse(body);
 29       if (json.errors && json.errors.length > 0) {
 30         errors = json.errors;
 31       } else {
 32         data = Swagger.toData(json.data);
 33       }
 34     }
 35     return {
 36       data:data,
 37       errors:errors
 38     };
 39   };
 40 };
 41 
 42 /**
 43  * Creates a new BackChat.io client.
 44  *
 45  * @class
 46  *
 47  * The BackchatClient provides a client for BackChat.io. The client emits the following events:
 48  * <ul>
 49  * <li>'ready' when the client is initialized and ready to be used</li>
 50  * <li>'error' when the client failed to initialize properly</li>
 51  * </ul>
 52  *
 53  * Usage:
 54  *
 55  * <pre>
 56  * var client = new backchat.BackchatClient('<api_key>');
 57  * client.on('ready', function() {
 58  *   client.api.account.userDetails();
 59  * });
 60  * </pre>
 61  *
 62  * @augments events.EventEmitter
 63  *
 64  * @param {String|Object} options a valid API key or an options object
 65  * @param {String} options.apiKey a valid API key
 66  * @param {String} [options.url] the api URL
 67  *
 68  * @property {Object} api the api object for working with the BackChat.io API 
 69  *
 70  * @throws {Error} if the options argument is undefined or invalid
 71  */
 72 var BackchatClient = function(options) {
 73   events.EventEmitter.call(this);
 74   if (typeof options === 'undefined') {
 75     throw new Error('undefined is not a valid apiKey or options object.');
 76   }
 77   if (typeof options === 'string') {
 78     options = {apiKey: options};
 79   }
 80   if (typeof options === 'object') {
 81     options.url = options.url || BackchatApiUrl;
 82   } else {
 83     throw new Error('options is not a valid options object.');
 84   }
 85   var self = this;
 86   Swagger.open({url: options.url,
 87     authHeaders: {'Authorization':'Backchat '+options.apiKey},
 88     responseHandler: function(op) { return new ResponseHandler(op); }},
 89     function(client) {
 90       self.api = client;
 91       self.emit('ready');
 92     });
 93 };
 94 
 95 util.inherits(BackchatClient, events.EventEmitter);
 96 
 97 module.exports.ResponseHandler = ResponseHandler;
 98 module.exports.BackchatClient = BackchatClient;
